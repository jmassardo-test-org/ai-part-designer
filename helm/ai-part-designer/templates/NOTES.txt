IMPORTANT: Before the application can function properly, you must create secrets:

  kubectl create secret generic backend-secrets \
    --namespace={{ .Release.Namespace }} \
    --from-literal=postgres-user=<your-user> \
    --from-literal=postgres-password=<your-password> \
    --from-literal=anthropic-api-key=<your-api-key> \
    --from-literal=secret-key=<your-secret-key> \
    --from-literal=access-key-id=<minio-access-key> \
    --from-literal=secret-access-key=<minio-secret-key>

  kubectl create secret generic celery-worker-secrets \
    --namespace={{ .Release.Namespace }} \
    --from-literal=access-key-id=<minio-access-key> \
    --from-literal=secret-access-key=<minio-secret-key>
